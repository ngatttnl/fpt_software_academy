const slider = document.querySelector('.slider');
let currentSlide = 0;

function nextSlide() {
    currentSlide = (currentSlide + 1) % 3;
    slider.style.transform = `translateX(-${currentSlide * 100}%)`;
}

setInterval(nextSlide, 3000);
